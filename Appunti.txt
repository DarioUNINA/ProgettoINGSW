Query asse y: 
select count (*) from ordine o join tavolo t on t.id_tavolo = o.id_tavolo 
where (username_cameriere = :cameriere and data > :dataFrom and data < :dataTo)
group by data

Query asse x:
select data
from ordine o join tavolo t on t.id_tavolo = o.id_tavolo
where (username_cameriere = :cameriere and data > :dataFrom and data < :dataTo)

Query guadagno totale:
select sum(quantita*prezzo) from ((ordine o join tavolo t on t.id_tavolo = o.id_tavolo) 
join ordinazione ord on ord.id_ordine = o.id_ordine)
join elemento e on e.id_elemento = ord.id_elemento
where (username_cameriere = :cameriere and data > :dataFrom and data < :dataTo)

Query ordini totali:
select count(id_ordine)
from ordine o join tavolo t on t.id_tavolo = o.id_tavolo
where (username_cameriere = :cameriere and data > :dataFrom and data < :dataTo)
